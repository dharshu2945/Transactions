 CREATE TABLE TRANSFER_TBL(
          Transaction_ID NUMBER(4) PRIMARY KEY,
          Account_Number VARCHAR2(10),
          Beneficiary_account_number VARCHAR2(10),
          Transaction_Date DATE,
          Transaction_Amount NUMBER(10,2)
          CONSTRAINT Account_Number_fk FOREIGN KEY REFERENCES ACCOUNT_TBL(Account_Number),
          CONSTRAINT Beneficiary_account_number_fk FOREIGN KEY REFERENCES ACCOUNT_TBL(Account_Number),
          );
          
create sequence transactionId_seq
START WITH 1000
MINVALUE 1000
MAXVALUE 9999
INCREMENT BY 1;
